const modeBtn = document.getElementById("mode-btn");
const clearBtn = document.getElementById("clear-btn");

let isFilling = false;

function colorChange(event) {
  ctx.strokeStyle = event.target.value;
  ctx.fillStyle = event.target.value;
}
         
function onModeClick(event) {
  if (isFilling) {
    isFilling = false;
    modeBtn.innerText = "Fill";
  } else {
    isFilling = true;
    modeBtn.innerText = "Draw";
  }
}

function onCanvasClick() {
  if (isFilling) {
    ctx.fillRect(0, 0, 800, 800);
  }
}
modeBtn.addEventListener("click", onModeClick);
canvas.addEventListener("click", onCanvasClick);

html에서 Fill 버튼을 만들고 js에서 변수 설정 id : mode-btn
isFilling 를 업데이트 가능한 변수로 만듦 (상태확인용)
버튼이 클릭되면 onModeClick이 실행되도록 이벤트 생성, IF로 isFilling가 true면
false로 변경 후, 텍스트를 Fill로 변경함. 아니라면 true로 변경후, Draw로 맞춤.

canvas에 클릭 이벤트를 추가, isFilling이 true라면 0,0,800,800만큼 채움

정리

isFilling으로 채울 지 말지를 결정함. if 조건문을 통해 true, false시 실행될 함수들을 설정, 결국 중요한건 onCanvasClick 함수
