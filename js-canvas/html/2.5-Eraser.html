const eraserBtn = document.getElementById("eraser-btn");

const CANVAS_WIDTH = 800;
const CANVAS_HEIGHT = 800;

localStorage.setItem("FillStyle", "#ffffff");

let isEraser = false;


function onEraserClick() {
  if (isEraser) {
    isEraser = false;
    eraserBtn.innerText = "Eraser";
  } else {
    isEraser = true;
    eraserBtn.innerText = "Active";
    ctx.strokeStyle = localStorage.getItem("FillStyle");
  }
}

eraserBtn.addEventListener("click", onEraserClick);

색을 지정하고 Fill을 할 경우 해당 색상 코드를 localStorage에 저장하고 eraser를 활성화 할 경우
localStorage에서 FillStyle 값을 불러와 strokeStyle에 지정한다. 기본 값은 white

localStorage를 사용하지 않는 다른 방법
